Para hacer los thumbnails de los mapitas:

import selenium, time
from selenium import webdriver
browser = webdriver.Firefox()
browser.get('http://localhost:8000')
# selectar todos los links a los mapas
links = browser.find_elements_by_css_selector('a[href^="#"]')
for l in links:
  l.click()
  time.sleep(0.25) # por si acaso
  browser.save_screenshot(l.get_attribute('href')[1:])

browser.quit()
